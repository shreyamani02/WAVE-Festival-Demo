function testButton() {

  var range = prompt("The number you will guess is from 1 to");

  alert("You will only get about 10 guesses for this guessing game!");

  var randomnumber = Math.floor(Math.random() * range) + 1;

  console.log(randomnumber);

  var guess = prompt("What is your guess?");

  var totalguesses = 0;

  while(totalguesses<11) {

    if (guess < randomnumber) {

      guess = prompt("This guess was too low. Please guess another value.")

    }

    if (guess > randomnumber) {

      guess = prompt("This guess is too high. Please input another number.")

    }  

    if (guess == randomnumber) {

      document.getElementById("demo").innerHTML = "Congragulations! You have guessed the number.";

      break;

    }

    totalguesses = totalguesses + 1;

  }

  if(totalguesses == 11);

    document.getElementById("demo").innerHTML = "Oops. You didn't get it this time. Try again soon!";

  }
